### onCheckedChange function

<Tabs items={["Preview", "Code"]}>
    <Tab className={"flex justify-center"} value="Preview">
        <SwiychEvent />
    </Tab>
    <Tab className={"-mt-8"} value="Code">
```tsx
// [!code word:onCheckedChange]
"use client";

import React, { useState } from "react";
import { Switch } from "ruru-ui/components/switch";

const SwiychDemo = (): React.ReactNode => {
  const [value, setValue] = useState<boolean>(false); // [!code highlight]
  return (
    <div className="flex flex-col justify-center items-center">
      <div className="flex items-center space-x-2">
        <Switch
          onCheckedChange={(e) => setValue(e)} // [!code highlight]
          id="toggle-animation"
        />
        <label htmlFor="toggle value"> toggle value </label>
      </div>
      <p>{String(value)}</p>
    </div>
  );
};

export default SwiychDemo;
```
